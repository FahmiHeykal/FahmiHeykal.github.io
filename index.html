// Handle koneksi WebSocket
func handleConnections(w http.ResponseWriter, r *http.Request) {
    ws, err := upgrader.Upgrade(w, r, nil)
    if err != nil {
        log.Println("Error upgrading connection:", err)
        return
    }
    defer ws.Close()

    mutex.Lock()
    clients[ws] = true
    mutex.Unlock()

    for {
        var msg Message
        err := ws.ReadJSON(&msg)
        if err != nil {
            log.Println("Error membaca pesan:", err)
            mutex.Lock()
            delete(clients, ws)
            mutex.Unlock()
            break
        }
        broadcast <- msg
    }
}
